1 #include <stdio.h>
          2  #include <string.h>

          4  // Struktur untuk menyimpan informasi barang

         16          {3, "Odol", 7000},
         17          {4, "Sandal", 3000},

         19          // Tambahkan barang lain di sini
         20      };

         23      int total_barang = sizeof(barang)/sizeof(barang[0]); // total barang yang tersedia

         25      int i;
         26  
         27      // Inisialisasi jumlah barang menjadi 0
         28      for (i = 0; i < total_barang; i++) {
         29          jumlah_barang[i] = 0;
         30      }

         33  
         34      // Meminta nama pembeli

         36      fgets(nama_pembeli, sizeof(nama_pembeli), stdin);
         37      strtok(nama_pembeli, "\n"); // menghapus karakter newline dari nama pembeli

         39      // Menampilkan daftar barang yang tersedia
         40      printf("\nDaftar Barang yang Tersedia:\n");

         73                      if (jumlah_barang[i] > 3) {
         74                          diskon = subtotal * 0.10; // Diskon 10% jika jumlah produk melebihi 3
         75                      }
         76                      if (jumlah_barang[i] > 5) {
         77                          diskon = subtotal * 0.15; // Diskon 15% jika jumlah produk melebihi 5
         78                      }
         79                      total_diskon += diskon;
         80                      printf("%-15s | %-13.2f | %-10d | %-11.2f | %-10.2f\n", barang[i].nama, barang[i].harga, jumlah_barang[i], subtotal, diskon);
         81                  }

         85              printf("Total Harga   : %.2f\n", total_harga);

         88              float uang_dibayarkan;

         91                  scanf("%f", &uang_dibayarkan);

         93                      printf("Uang yang dibayarkan kurang dari total harga. Mohon masukkan kembali.\n");
         94                  }
         95              } while (uang_dibayarkan < total_harga);
         96  
         97              float kembalian = uang_dibayarkan - total_harga;
         98              printf("Uang kembalian          : %.2f\n", kembalian);
         99              printf("\n-------------------------------------------------\n");
        100  
        101              // Simpan struk belanja ke dalam file teks
        102              char filename[50];
        103              sprintf(filename, "struk_%s.txt", nama_pembeli);
        104              FILE *file;
        105              file = fopen(filename, "w");
        106              if (file != NULL) {
        107                  fprintf(file, "================= Struk Belanja =================\n");
        108                  fprintf(file, "Nama Pembeli   : %s\n", nama_pembeli);
        109                  fprintf(file, "Nama Produk    | Harga Satuan  | Jumlah     | Total Harga | Diskon    \n");
        110                  fprintf(file, "-------------------------------------------------\n");

        112                      if (jumlah_barang[i] > 0) {
        113                          float subtotal = barang[i].harga * jumlah_barang[i];
        114                          float diskon = 0;
        115                          if (jumlah_barang[i] > 3) {
        116                              diskon = subtotal * 0.10; // Diskon 10% jika jumlah produk melebihi 3
        117                          }
        118                          if (jumlah_barang[i] > 5) {
        119                              diskon = subtotal * 0.15; // Diskon 15% jika jumlah produk melebihi 5
        120                          }
        121                          fprintf(file, "%-15s | %-13.2f | %-10d | %-11.2f | %-10.2f\n", barang[i].nama, barang[i].harga, jumlah_barang[i], subtotal, diskon);
        122                      }
        123                  }
        124                  fprintf(file, "-------------------------------------------------\n");
        125                  fprintf(file, "Total Harga   : %.2f\n", total_harga);
        126                  fprintf(file, "Total diskon  : %.2f\n", total_diskon);
        127                  fprintf(file, "Uang yang dibayarkan    : %.2f\n", uang_dibayarkan);
        128                  fprintf(file, "Uang kembalian          : %.2f\n", kembalian);
        129                  fprintf(file, "\n-------------------------------------------------\n");

        143              if (barang[i].kode == kode) {
        144                  jumlah_barang[i] += jumlah;
        145                  total_belanja += barang[i].harga * jumlah;
        146                  break;
        147              }
        148          }
        149          if (i == total_barang) {
        150              printf("Kode barang tidak valid.\n");
        151          }
        152      } while (1);
        153  
        154      return 0;
